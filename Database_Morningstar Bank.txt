CREATE TABLE  AccountMaster  (
	 Account_No  NUMBER(13, 0) NOT NULL,
	 Title  VARCHAR2(10) NOT NULL,
	 First_Name  VARCHAR2(20) NOT NULL,
	 Last_Name  VARCHAR2(20) NOT NULL,
	 Father_Name  VARCHAR2(20) NOT NULL,
	 Mobile  NUMBER(10, 0) NOT NULL,
	 Email  VARCHAR2(30) NOT NULL,
	 Adhar  NUMBER(12, 0) NOT NULL,
	 Date_Of_Birth  VARCHAR2(255) NOT NULL,
	 Address  VARCHAR2(50) NOT NULL,
	 City  VARCHAR2(20) NOT NULL,
	 State  VARCHAR2(20) NOT NULL,
	 Pincode  NUMBER(6, 0) NOT NULL,
	 Account_Type  VARCHAR2(20) NOT NULL,
	 Occupation  VARCHAR2(255) NOT NULL,
	 Gross_Anual_Income  INT NOT NULL,
	 Netbanking  VARCHAR2(10) NOT NULL,
	constraint ACCOUNTMASTER_PK PRIMARY KEY ( Account_No ));

drop sequence  ACCOUNT_NO_SEQ ;
CREATE sequence  ACCOUNT_NO_SEQ  start with 1800000000001 increment by 1;

CREATE TABLE  InternetBanking  (
	 Internet_Banking_Id  INT NOT NULL,
	 Login_Password  VARCHAR2(255) NOT NULL,
	 Transaction_Password  VARCHAR2(255) NOT NULL,
	 Secret_Question  VARCHAR2(255) NOT NULL,
	 Secret_Answer  VARCHAR2(255) NOT NULL,
	 Transaction_Email  VARCHAR2(255) NOT NULL,
	 Account_No  NUMBER(13, 0) NOT NULL,
	constraint INTERNETBANKING_PK PRIMARY KEY ( Internet_Banking_Id ));


/
CREATE TABLE  AccountDetails  (
	 Balance  INT NOT NULL,
	 Opening_Date  VARCHAR2(255) NOT NULL,
	 Account_No  INT NOT NULL);


/
CREATE TABLE  TransactionMaster  (
	 Reference_Id  INT NOT NULL,
	 Payee_Account_No  INT NOT NULL,
	 Transfer_Amount  FLOAT NOT NULL,
	 Transfer_Mode  VARCHAR2(255) NOT NULL,
	 Remarks  VARCHAR2(255) NOT NULL,
	 Internet_Banking_Id  INT NOT NULL,
	constraint TRANSACTIONMASTER_PK PRIMARY KEY ( Reference_Id ));

drop sequence  REFERENCE_ID_SEQ ;
CREATE sequence  REFERENCE_ID_SEQ ;



/
CREATE TABLE  BeneficiaryDetails  (
	 Payee_Account_No  INT NOT NULL,
	 Account_No  INT NOT NULL,
	 Name  VARCHAR2(255) NOT NULL,
	 Nick_Name  VARCHAR2(255) NOT NULL,
	constraint BENEFICIARYDETAILS_PK PRIMARY KEY ( Payee_Account_No ));
